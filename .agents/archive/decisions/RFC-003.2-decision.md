## Schema Decisions (canonical — executor must follow exactly)

### CaseIntake uses `causante` group (not flat fields)

Rationale: `Cases.causante` is already a group. Using the same structure on `CaseIntake` makes the conversion a direct group copy (`Case.causante = CaseIntake.causante`) — no field-by-field mapping needed, no mismatch risk.

### `parentesco` option values — standard

Use these exact values across both `CaseIntake.parentescoConCausante` and `Heirs.parentesco`:

```
{ label: 'Hijo/a',       value: 'hijo'       }
{ label: 'Hermano/a',    value: 'hermano'    }
{ label: 'Cónyuge',      value: 'conyuge'    }
{ label: 'Nieto/a',      value: 'nieto'      }
{ label: 'Sobrino/a',    value: 'sobrino'    }
{ label: 'Padre/Madre',  value: 'padre_madre'}
{ label: 'Otro',         value: 'otro'       }
```

No slashes, no accents in values. Labels may use full display text.

### Representation model

A deceased intermediate heir (e.g., a son who died before the causante) IS entered as a `Heirs` record with `esFallecido: true`. Their children (the representatives) are entered as separate `Heirs` records with `esRepresentante: true` and `herederoOriginal` pointing to the deceased parent's record.

- `esFallecido`: only a lawyer/admin can set this to `true` (field-level AC on update).
- `esRepresentante` and `herederoOriginal`: client CAN set these (they know their own family situation).
- `representacionValidada`: only a lawyer/admin can set this (field-level AC on read + update). This is the legal gate — no document generation uses an unvalidated representative.

DocumentChecklist note (OUT OF SCOPE for this RFC — flagged for RFC future phase):  
When phase advances to 2+, the hook must generate DIFFERENT documents per heir type:  
- Normal alive heir → Registro civil de nacimiento + Cédula  
- `esFallecido: true` → Registro civil de defunción  
- `esRepresentante: true` → Registro civil de nacimiento proving lineage + Cédula  

### Assets `investigacion.resultado`

Points to `media` collection (Payload's built-in upload collection). Will be redirected to `documents` collection when Documents is implemented in a future RFC. Field name and shape stay the same.

### `valorEstimadoTotal`

Client self-estimates the total market value of all estate assets combined (in COP). Used by Paola as reference for notarial fee calculation and by Germán for CRM scoring of draft intakes. This field is NOT in `COLLECTIONS.md` — TASK 9 adds it there.

### Conversion trigger

For this RFC: a REST endpoint `POST /api/succession/convert-intake` (admin/lawyer only). This same utility function will be called by the Wompi webhook in RFC-008 — do NOT duplicate the logic.

```typescript
import type { CollectionConfig, CollectionBeforeChangeHook, Where } from 'payload'
import { isAdminUser, isMemberUser } from '@/lib/auth/typeGuards'

const validateOnSubmit: CollectionBeforeChangeHook = async ({ data, originalDoc }) => {
  const incoming = data as Record<string, unknown>
  const status = incoming.status ?? originalDoc?.status

  if (status !== 'submitted') return data

  const causante = (incoming.causante ?? originalDoc?.causante) as Record<string, unknown> | undefined
  const required = [
    causante?.nombre,
    causante?.cedula,
    causante?.fechaFallecimiento,
    causante?.ciudadFallecimiento,
    incoming.tierElegido ?? originalDoc?.tierElegido,
    incoming.herederosEstimados ?? originalDoc?.herederosEstimados,
    incoming.bienesDescripcion ?? originalDoc?.bienesDescripcion,
    incoming.acuerdoHerederos ?? originalDoc?.acuerdoHerederos,
  ]

  const missing = required.some(v => v === undefined || v === null || v === '')
  if (missing) {
    throw new Error('Todos los campos requeridos deben estar completos antes de enviar el intake.')
  }

  return data
}
```

```typescript
import type { CollectionConfig, Where } from 'payload'
import { isAdminUser, isMemberUser } from '@/lib/auth/typeGuards'

const isAdminOrLawyer = (user: unknown): boolean =>
  isAdminUser(user) ||
  (isMemberUser(user) && user.role === 'abogado' && user.isActive !== false)

export const Heirs: CollectionConfig = {
  slug: 'heirs',
  access: {
    read: async ({ req }) => {
      const user = req.user
      if (!user) return false
      if (isAdminUser(user)) return true
      if (!isMemberUser(user)) return false
      if (user.role === 'abogado' && user.isActive !== false) return true

      // Client: only heirs belonging to their cases
      const cases = await req.payload.find({
        collection: 'cases',
        where: { responsable: { equals: user.id } },
        select: { id: true },
        limit: 100,
      })
      const ids = cases.docs.map((c) => c.id)
      if (ids.length === 0) return false
      return { case: { in: ids } } as Where
    },
```

#### TASK 5 — Create conversion utility

**File:** `src/payload/collections/succession/CaseIntake/utils/convertIntakeToCase.ts`

```typescript
import type { Payload } from 'payload'

/**
 * Converts a submitted CaseIntake into a live Case.
 * Called by the admin endpoint (RFC-003.2) and the Wompi webhook (RFC-008).
 *
 * Field mapping:
 *   CaseIntake.causante.nombre              → Case.causante.nombre
 *   CaseIntake.causante.cedula              → Case.causante.cedula
 *   CaseIntake.causante.fechaFallecimiento  → Case.causante.fechaFallecimiento
 *   CaseIntake.causante.ciudadFallecimiento → Case.causante.ciudadFallecimiento
 *   CaseIntake.tierElegido                  → Case.tier  (default: 'estandar')
 *   CaseIntake.member                       → Case.responsable
 *   CaseIntake.tieneTestamento              → Case.tieneTestamento
 *   CaseIntake.acuerdoHerederos             → Case.acuerdoHerederos
 *   CaseIntake.id                           → Case.caseIntake  (reverse link)
 *   CaseIntake.appointment                  → Case.appointment (if present)
 *
 * After creation:
 *   CaseIntake.status  ← 'paid'
 *   CaseIntake.case    ← Case.id
 *
 * Credit application (RFC-003.3): deferred — do NOT touch Appointments here.
 */
export async function convertIntakeToCase(intakeId: string, payload: Payload): Promise<string> {
  const intake = await payload.findByID({
    collection: 'case-intakes',
    id: intakeId,
    overrideAccess: true,
  })

  if (!intake) throw new Error(`CaseIntake not found: ${intakeId}`)
  if (intake.status === 'paid') throw new Error(`CaseIntake ${intakeId} is already converted`)

  const memberId =
    typeof intake.member === 'object' && intake.member !== null
      ? (intake.member as { id: string }).id
      : (intake.member as string)

  const appointmentId = intake.appointment
    ? typeof intake.appointment === 'object'
      ? (intake.appointment as { id: string }).id
      : (intake.appointment as string)
    : undefined

  const causante = intake.causante as {
    nombre?: string
    cedula?: string
    fechaFallecimiento?: string
    ciudadFallecimiento?: string
  } | undefined

  const newCase = await payload.create({
    collection: 'cases',
    data: {
      causante: {
        nombre: causante?.nombre ?? '',
        cedula: causante?.cedula ?? '',
        fechaFallecimiento: causante?.fechaFallecimiento,
        ciudadFallecimiento: causante?.ciudadFallecimiento ?? '',
      },
      tier: (intake.tierElegido as 'estandar' | 'premium' | 'elite') ?? 'estandar',
      responsable: memberId,
      tieneTestamento: intake.tieneTestamento ?? false,
      acuerdoHerederos: (intake.acuerdoHerederos as 'si' | 'no_sabe') ?? 'si',
      caseIntake: intakeId,
      ...(appointmentId ? { appointment: appointmentId } : {}),
      status: 'active',
      currentPhase: 0,
    },
    overrideAccess: true,
  })

  await payload.update({
    collection: 'case-intakes',
    id: intakeId,
    data: {
      status: 'paid',
      case: newCase.id,
    },
    overrideAccess: true,
  })

  return newCase.id
}
```

#### TASK 6 — Create conversion API endpoint

**File:** `src/app/api/succession/convert-intake/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { getPayload } from 'payload'
import configPromise from '@payload-config'
import { isAdminUser, isMemberUser } from '@/lib/auth/typeGuards'
import { convertIntakeToCase } from '@/payload/collections/succession/CaseIntake/utils/convertIntakeToCase'

export async function POST(req: NextRequest): Promise<NextResponse> {
  const payload = await getPayload({ config: configPromise })

  const { user } = await payload.auth({ headers: req.headers })

  const authorized =
    isAdminUser(user) ||
    (isMemberUser(user) && user.role === 'abogado' && user.isActive !== false)

  if (!authorized) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 403 })
  }

  let intakeId: string
  try {
    const body = await req.json()
    intakeId = body.intakeId
  } catch {
    return NextResponse.json({ error: 'Invalid JSON body' }, { status: 400 })
  }

  if (!intakeId) {
    return NextResponse.json({ error: 'intakeId is required' }, { status: 400 })
  }

  try {
    const caseId = await convertIntakeToCase(intakeId, payload)
    return NextResponse.json({ caseId }, { status: 201 })
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Unknown error'
    return NextResponse.json({ error: message }, { status: 500 })
  }
}
```
